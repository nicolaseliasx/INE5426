def int calcular_fatorial(int n) {
    int resultado;
    int i;
    resultado = 1;
    if (n > 1) {
        for (i = 2; i <= n; i = i + 1) {
            resultado = resultado * i;
        }
    }
    return resultado;
}

def void processar_vetor(int v[10]) {
    int k;
    int soma_par;
    soma_par = 0;
    print("Processando o vetor...");
    for (k = 0; k < 10; k = k + 1) {
        if (v[k] % 2 == 0) {
            soma_par = soma_par + v[k];
        }
    }
    print("Soma dos pares no vetor: ", soma_par);
}
